let beforeRun = async (req, res, next) => {
  try {
    let { _id } = req.params;       // /chat/:_id
    let { pass } = req.body;        // input name="pass"

    let chatUser = await Chat.findById(_id);

    if (!chatUser) {
      return res.send("User not found");
    }

    let { password } = chatUser;

    let isMatch = await bcrypt.compare(pass, password);

    if (isMatch) {
      next();  // correct password â†’ go ahead
    } else {
      return res.send("Password not match");
    }

  } catch (err) {

    res.send("Error occurred");
  }
};
